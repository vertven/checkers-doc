102 // `index` is index of the cell in `Positions`
103 var index = 0;
104
105 // `y` starts at BoardSize - 1 because of the discontinuity of the origin.
106 // `Positions`'s origin is top-left while the graphical origin is bottom left
107 for (var y = BoardSize - 1; y >= 0; y--)
108 {
109 // Cell separator
110 Console.Write("|");
111
112 for (var x = 0; x < BoardSize; x++)
113 {
114 BasePiece currPiece = null;
115 var reachableCell = y % 2 == 0 && x % 2 == 0 || y % 2 != 0 && x % 2 != 0;
116
117 // Only if the cell is reachable, update `currentPiece` with the piece on the cell
118 if (reachableCell)
119 currPiece = Positions[index++].CurrentPiece;
120
121 // `currentPiece` can be null. In that case the foreground color does not matter.
122 Console.ForegroundColor = currPiece?.Color == PlayerColor.White
123 ? ConsoleColor.DarkRed
124 : ConsoleColor.DarkBlue;
125
126 // If there is no piece then consider this cell as unreachable thus white
127 Console.BackgroundColor = reachableCell ? ConsoleColor.Black : ConsoleColor.White;
128
129 // Print the piece. If null print spaces
130 switch (currPiece)
131 {
132 case King or Man:
133 Console.Write(currPiece is King ? " K " : " M ");
134 break;
135 default:
136 Console.Write(" ");
137 break;
138 }
