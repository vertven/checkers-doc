70 // `y` starts at BoardSize - 1 because of the discontinuity of the origin.

72 for (var y = BoardSize - 1; y >= 0; y--)
73 {
74 // Cell separator
75 Console.Write("|");
76
77 for (var x = 0; x < BoardSize; x++)
78 {
79 BasePiece currPiece = null;
80 var reachableCell = y % 2 == 0 && x % 2 == 0 || y % 2 != 0 && x % 2 != 0;
81
82 // Only if the cell is reachable, update `currentPiece` with the piece on the cell
83 if (reachableCell)
84 currPiece = Positions[index++].CurrentPiece;
85
86 // `currentPiece` can be null. In that case the foreground color does not matter.
87 Console.ForegroundColor = currPiece?.Color == PlayerColor.White
88 ? ConsoleColor.DarkRed
89 : ConsoleColor.DarkBlue;
90
91 // If there is no piece then consider this cell as unreachable thus white
92 Console.BackgroundColor = reachableCell ? ConsoleColor.Black : ConsoleColor.White;
93
94 // Print the piece. If null print spaces
95 switch (currPiece)
96 {
97 case King or Man:

100 default:
