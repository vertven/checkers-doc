95 public List<PathObject> CheckPathing(int pos, List<BasePiece> prevEnemies)
96 {
97 var capturing = prevEnemies is not null;
98 var res = new List<PathObject>();
99
100 // Try to create a path in all the piece's possible diagonals
101 foreach (var dir in DirectionHelper.GetDirections(this))
102 {
103 // Get possible paths (if any) in this direction. If none was found, try another direction
104 var possiblePaths = CreateCellPath(pos, dir, prevEnemies);
105 if (possiblePaths is null)
106 continue;
107
108 // `foundEnemies` gathers all the paths with enemies to check if you have encountered enemies
109 var foundEnemies = possiblePaths.FindAll(pathObj => pathObj.Enemies.Count > 0);
110
111 switch (foundEnemies.Count > 0)
112 {
113 // If you did not encounter enemies but you should have (prevEnemies is not null or you encountered
114 // enemies in another direction) continue
115 case false when capturing:
116 continue;
117 // Else set `possiblePaths` to be all the "capture move" found and set `capturing` to true
118 case true:
119 possiblePaths = foundEnemies;
120 capturing = true;
121 break;
122 }
123
124 // Add `possiblePaths` to `res`
125 res.AddRange(possiblePaths);
126 }
127
128 return res;
129 }

142 private List<PathObject> CreateCellPath(int pos, Direction direction, List<BasePiece> enemies)
143 {
144 // Create a temporary variable to store the position of the piece
145 var currPos = pos;
146
147 // Create the resulting list of PathObjects
148 // and a temporary list of Cells initialised with the cell the piece is currently on
149 List<PathObject> res = null;
150 var path = new List<Cell> {Board.Positions[currPos - 1]};
151
152 // If `enemies` is null then initialise the list, otherwise clone it
153 enemies = enemies is null ? new List<BasePiece>() : enemies.GetRange(0, enemies.Count);
154
155 // Loop until no moves left
156 // If `Movement` is greater than one then the piece is a king and can only move more than 1 cell if and only
157 // if there is a enemy piece to capture on the path
158 for (var i = 0; i < ScanDistance; i++)
159 {
160 // Update the current position in the direction specified
161 // The function returns the new position or -1 if out of bounds
162 currPos = DirectionHelper.GetCellPosInDirection(currPos, direction);
163
164 // Gets the cell state (if `currPos` == -1 => out of bounds)
165 var cellState = currPos == -1 ? Cell.CellState.OutOfBounds : Board.GetCellState(currPos, Color);
166
167 // If the cell is an enemy, try to take it
168 if (cellState == Cell.CellState.Enemy)
169 {
170 // Checks if you already took the enemy piece. If uou did, a path in this direction is not viable
171 var enemy = Board.Positions[currPos - 1].CurrentPiece;
172 if (enemies.Contains(enemy))
173 return res;
174
175 // Update the current position in the direction specified
176 // The function returns the new position or -1 if out of bounds
177 currPos = DirectionHelper.GetCellPosInDirection(currPos, direction);
178
179 // If `currPos` == 1 then cell is out of bounds, returns `res`
180 // Else if the cell is not free then impossible move, return `res`
181 if (currPos == -1 || Board.GetCellState(currPos, Color) != Cell.CellState.Free)
182 return res;
183
184 // Adds the enemy cell to `_enemyEncountered`
185 enemies.Add(enemy);
186
187 // Update the possible path with the cell after the enemy
188 path.Add(Board.Positions[currPos - 1]);
189
190 // Reset the previously found (if piece is a king and if any) simple move because if a capture is
191 // possible you must take it. If `res` was null, create it
192 res?.Clear();
193 res ??= new List<PathObject>();
194
195 // Create a copy of the current path
196 var pathCopy = path.GetRange(0, path.Count);
197
198 // Check if you can capture more pieces from here
199 var nextPaths = CheckPathing(currPos, enemies);
200
201 // If you could capture more pieces after the initial capture
202 if (nextPaths.Count != 0)
203 {
204 // For each additional path found, concatenate their path with the current possible path and add
205 // them to `res`
206 res.AddRange(nextPaths.Select(pathObj =>
207 new PathObject(pathCopy.Concat(pathObj.Path).ToList(), pathObj.Enemies)));
208 }
209 // If you could not add the current path and a copy of the enemies encountered to `res`
210 else
211 res.Add(new PathObject(pathCopy, enemies.GetRange(0, enemies.Count)));
212
213 // You have captured an enemy and checked if you could capture more, now you can just return `res`
214 // as there is nothing else to do in this direction
215 return res;
216 }
217
218 // If the cell is not free you cannot build a path in this direction
219 if (cellState != Cell.CellState.Free)
220 return res;
221
222 // You have fround a free cell, add it to the current possible path list
223 path.Add(Board.Positions[currPos - 1]);
224
225 // If you are a king and that you have already used your simple move, you can now only add a possible
226 // path to the result if and only if there is an enemy to capture on that path
227 if (ScanDistance > 1 && i > 0)
228 continue;
229
230 // Create `res` if null then add to `res` a new PathObject with a copy of the current path and a copy of
231 // the list of enemies encountered
232 res ??= new List<PathObject>();
233 res.Add(new PathObject(path.GetRange(0, path.Count), enemies.GetRange(0, enemies.Count)));
234 }
235
236 return res;
237 }

265 private PathObject RecreatePath(int pos, Cell target, List<BasePiece> previousEnemies)
266 {
267 PathObject res = null;
268 var distance = 0;
269
270 foreach (var path in CheckPathing(pos, previousEnemies)
271 .FindAll(pathTuple => pathTuple.Path.Contains(target)))
272 {
273 var newDistance = path.Path.Count + path.Enemies.Count;
274 if (newDistance <= distance)
275 continue;
276
277 distance = newDistance;
278 res = path;
279 }
280
281 return res;
282 }

346 public virtual bool Move(PathObject pathObj, bool simulation = false)
347 {
348 // If the path is null or if you could have captured but the move is a "simple move" then return
349 if (pathObj?.Path is null || CheckCouldHaveCaptured(pathObj.Enemies.Count != 0))
350 return false;
351
352 // Capture all the pieces on cells in `pathObj.Enemies`
353 foreach (var enemy in pathObj.Enemies)
354 enemy.Kill(simulation);
355
356 // Update the `CurrentPiece` of the begin and end cell
357 CurrentCell.CurrentPiece = null;
358 CurrentCell = pathObj.Path.Last();
359 CurrentCell.CurrentPiece = this;
360 return true;
361 }

375 public virtual void RevertMove(PathObject pathObj)
376 {
377 // If the path is null then return
378 if (pathObj?.Path is null)
379 return;
380
381 // Un-kill the piece by replacing it on the correct cell and re-adding it to the piece manager
382 foreach (var enemy in pathObj.Enemies)
383 {
384 enemy.CurrentCell.CurrentPiece = enemy;
385 Board.PieceManager.AddPiece(enemy);
386 }
387
388 // Update the `CurrentPiece` of the begin and end cell
389 CurrentCell.CurrentPiece = null;
390 CurrentCell = pathObj.Path.First();
391 CurrentCell.CurrentPiece = this;
