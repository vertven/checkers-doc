45 public static void Decrypt(PieceManager pieceManager, Board board, string fenCode)
46 {
47 // Split the string by row and check if there is the exact number of rows
48 var rows = fenCode.Split('/');
49 if (rows.Length != Board.BoardSize)
50 throw new ArgumentException("Invalid board size");
51
52 // FEN notations starts from the lower left corner and goes to the upper right corner row by row
53 // In order to match our representation of the board we need to reverse the array `rows` that way we have a
54 // left to right and top to bottom representation of the board
55
56 // `index` is the index of the cell in `board.Positions`
57 var index = 0;
58 foreach (var row in rows.Reverse())
59 {
60 // `rowPieceCounter` makes sure that there is EXACTLY row's length pieces per row (pieces can be null)
61 var rowPieceCounter = 0;
62
63 // Iterate through this row's code segment
64 foreach (var c in row)
65 {
66 // If you already placed 4 pieces (pieces could have been null), throw an exception
67 if (rowPieceCounter >= Board.RowLength)
68 throw new ArgumentException("Wrong number of pieces in a single row");
69
70 // Temporary variables for code readability
71 var tmpChar = c;
72 var pieceColor = PlayerColor.White;
73
74 // If `c` is a digit it then indicates how many cells to skip
75 // (hence `CurrentPiece` for those cells will be null)
76 if (char.IsDigit(c))
77 {
78 // Calculates the amount to skip. If greater than for or less than 0 (should not happen haha)
79 // then throw and exception
80 var amount = c - '0';
81 if (amount is <= 0 or > Board.RowLength)
82 throw new ArgumentException("Invalid empty space amount");
83
84 // Update the index and the piece counter
85 index += amount;
86 rowPieceCounter += amount;
87
88 // Skips to the next character
89 continue;
90 }
91
92 // If `c` is an upper character, set the new piece's color to black and lower it
93 if (char.IsUpper(c))
94 {
95 tmpChar = char.ToLower(c);
96 pieceColor = PlayerColor.Black;
97 }
98
99 // At this point you have a lowercase char. If it is not an 'm' or and 'k' it is not recognized and
100 // thus an exception is thrown
101 if (tmpChar is not ('m' or 'k'))
102 throw new ArgumentException("Invalid character");
103
104 // Create the new piece with its appropriate type and color
105 BasePiece newPiece = tmpChar is 'm' ? new Man(pieceColor) : new King(pieceColor);
106
107 // Put the new piece in the board and in the pieceManager's lists
108 pieceManager.AddPiece(newPiece);
109 newPiece.Place(board.Positions[index]);
110
111 // Update the index the amount of pieces in the current row
112 index++;
113 rowPieceCounter++;
114 }
115
116 // If at the end of the row's code segment you did not add a total of 4 pieces (null included) an
117 // exception must be thrown
118 if (rowPieceCounter != Board.RowLength)
119 throw new ArgumentException("Wrong number of pieces in a single row");
120 }
121 }
