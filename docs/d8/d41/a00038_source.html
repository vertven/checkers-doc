29 // Gets the board position of the next cell from a given board position in a given direction
30 public static int GetCellPosInDirection(int pos, Direction direction)
31 {
32 // If `pos` is outside of bounds, return -1
33 if (pos is <= 0 or > Board.NbPositions)
34 return -1;
35
36 // Check if you are trying to get an impossible direction. If so return -1
37 switch (pos)
38 {
39 case <= Board.RowLength when direction is Direction.UpperLeft or Direction.UpperRight:
40 case >= Board.NbPositions - Board.RowLength + 1
41 when direction is Direction.LowerLeft or Direction.LowerRight:
42 case >= Board.RowLength and <= Board.NbPositions - Board.RowLength + 1
43 when pos % (Board.RowLength * 2) == Board.RowLength + 1 &&
44 direction is Direction.LowerLeft or Direction.UpperLeft ||
45 pos % (Board.RowLength * 2) == Board.RowLength &&
46 direction is Direction.LowerRight or Direction.UpperRight:
47 return -1;
48 }
49
50 // Return the according position
51 return pos + (int) direction - (pos - 1) / Board.RowLength % 2;
52 }
