119 // Cell separator

122 for (var x = 0; x < BoardSize; x++)
123 {
124 BasePiece currPiece = null;
125 var reachableCell = y % 2 == 0 && x % 2 == 0 || y % 2 != 0 && x % 2 != 0;
126
127 // Only if the cell is reachable, update `currentPiece` with the piece on the cell
128 if (reachableCell)
129 currPiece = Positions[index++].CurrentPiece;
130
131 // `currentPiece` can be null. In that case the foreground color does not matter.
132 Console.ForegroundColor = currPiece?.Color == PlayerColor.White
133 ? ConsoleColor.DarkRed

136 // If there is no piece then consider this cell as unreachable thus white
137 Console.BackgroundColor = reachableCell ? ConsoleColor.Black : ConsoleColor.White;
