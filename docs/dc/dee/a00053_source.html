57 public Cell.CellState GetCellState(int targetPosition, PlayerColor color)
58 {
59 // Check if target is out of bounds
60 if (targetPosition is <= 0 or > NbPositions)
61 return Cell.CellState.OutOfBounds;
62
63 // Get the cell at `targetPosition` (its index in `Positions` is one less than its board position)
64 // If there are no pieces on target, return free state
65 var targetCell = Positions[targetPosition - 1];
66 if (targetCell.CurrentPiece == null)
67 return Cell.CellState.Free;
68
69 // Compare the target's current piece's color with `color`
70 // If same then one of the color's pieces is on the cell, otherwise it's an enemy piece
71 return color == targetCell.CurrentPiece.Color ? Cell.CellState.Friendly : Cell.CellState.Enemy;

123 public void Print()
124 {
125 // `index` is index of the cell in `Positions`
126 var index = 0;
127
128 // `y` starts at BoardSize - 1 because of the discontinuity of the origin.
129 // `Positions`'s origin is top-left while the graphical origin is bottom left
130 for (var y = BoardSize - 1; y >= 0; y--)
131 {
132 // Cell separator
133 Console.Write("|");
134
135 for (var x = 0; x < BoardSize; x++)
136 {
137 BasePiece currPiece = null;
138 var reachableCell = y % 2 == 0 && x % 2 == 0 || y % 2 != 0 && x % 2 != 0;
139
140 // Only if the cell is reachable, update `currentPiece` with the piece on the cell
141 if (reachableCell)
142 currPiece = Positions[index++].CurrentPiece;
143
144 // `currentPiece` can be null. In that case the foreground color does not matter.
145 Console.ForegroundColor = currPiece?.Color == PlayerColor.White
146 ? ConsoleColor.DarkRed
147 : ConsoleColor.DarkBlue;
148
149 // If there is no piece then consider this cell as unreachable thus white
150 Console.BackgroundColor = reachableCell ? ConsoleColor.Black : ConsoleColor.White;
151
152 // Print the piece. If null print spaces
153 switch (currPiece)
154 {
155 case King or Man:
156 Console.Write(currPiece is King ? " K " : " M ");
157 break;
158 default:
159 Console.Write(" ");
160 break;
161 }
162
163 // Cell separator

165 Console.Write("|");
