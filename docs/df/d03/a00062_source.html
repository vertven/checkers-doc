103 private List<PathObject> CreateCellPath(int pos, Direction direction, List<BasePiece> enemies)
104 {
105 // Create a temporary variable to store the position of the piece
106 var currPos = pos;
107
108 // Create the resulting list of PathObjects
109 // and a temporary list of Cells initialised with the cell the piece is currently on
110 List<PathObject> res = null;
111 var path = new List<Cell> {Board.Positions[currPos - 1]};
112
113 // If `enemies` is null then initialise the list
114 enemies ??= new List<BasePiece>();
115
116 // Loop until no moves left
117 // If `Movement` is greater than one then the piece is a king and can only move more than 1 cell if and only
118 // if there is a enemy piece to capture on the path
119 for (var i = 0; i < ScanDistance; i++)
120 {
121 // Update the current position in the direction specified
122 // The function returns the new position or -1 if out of bounds
123 currPos = DirectionHelper.GetCellPosInDirection(currPos, direction);
124
125 // Gets the cell state (if `currPos` == -1 => out of bounds)
126 var cellState = currPos == -1 ? Cell.CellState.OutOfBounds : Board.GetCellState(currPos, Color);
127
128 // If the cell is an enemy, try to take it
129 if (cellState == Cell.CellState.Enemy)
130 {
131 // Checks if you already took the enemy piece. If uou did, a path in this direction is not viable
132 var enemy = Board.Positions[currPos - 1].CurrentPiece;
133 if (enemies.Contains(enemy))
134 return res;
135
136 // Update the current position in the direction specified
137 // The function returns the new position or -1 if out of bounds
138 currPos = DirectionHelper.GetCellPosInDirection(currPos, direction);
139
140 // If `currPos` == 1 then cell is out of bounds, returns `res`
141 // Else if the cell is not free then impossible move, return `res`
142 if (currPos == -1 || Board.GetCellState(currPos, Color) != Cell.CellState.Free)
143 return res;
144
145 // Adds the enemy cell to `_enemyEncountered`
146 enemies.Add(enemy);
147
148 // Update the possible path with the cell after the enemy
149 path.Add(Board.Positions[currPos - 1]);
150
151 // Reset the previously found (if piece is a king and if any) simple move because if a capture is
152 // possible you must take it. If `res` was null, create it
153 res?.Clear();
154 res ??= new List<PathObject>();
155
156 // Create a copy of the current path
157 var pathCopy = path.GetRange(0, path.Count);
158
159 // Check if you can capture more pieces from here
160 var nextPaths = CheckPathing(currPos, enemies);
161
162 // If you could capture more pieces after the initial capture
163 if (nextPaths.Count != 0)
164 {
165 // For each additional path found, concatenate their path with the current possible path and add
166 // them to `res`
167 res.AddRange(nextPaths.Select(pathObj =>
168 new PathObject(pathCopy.Concat(pathObj.Path).ToList(), pathObj.Enemies)));
169 }
170 // If you could not add the current path and enemies encountered to `res`
171 else
172 res.Add(new PathObject(pathCopy, enemies.GetRange(0, enemies.Count)));
173
174 // You have captured an enemy and checked if you could capture more, now you can just return `res`
175 // as there is nothing else to do in this direction
176 return res;
177 }
178
179 // If the cell is not free you cannot build a path in this direction
180 if (cellState != Cell.CellState.Free)
181 return res;
182
183 // You have fround a free cell, add it to the current possible path list
184 path.Add(Board.Positions[currPos - 1]);
185
186 // If you are a king and that you have already used your simple move, you can now only add a possible
187 // path to the result if and only if there is an enemy to capture on that path
188 if (ScanDistance > 1 && i > 0)
189 continue;
190
191 // Create `res` if null then add to `res` a new PathObject with a copy of the current path and the list
192 // of enemies encountered
193 res ??= new List<PathObject>();
194 res.Add(new PathObject(path.GetRange(0, path.Count), enemies.GetRange(0, enemies.Count)));
195 }
196
197 return res;
198 }
