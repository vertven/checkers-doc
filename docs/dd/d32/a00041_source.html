44 public static int GetCellPosInDirection(int pos, Direction direction)
45 {
46 // If `pos` is outside of bounds, return -1
47 if (pos is <= 0 or > Board.NbPositions)
48 return -1;
49
50 // Check if you are trying to get an impossible direction. If so return -1
51 switch (pos)
52 {
53 case <= Board.RowLength when direction is Direction.UpperLeft or Direction.UpperRight:
54 case >= Board.NbPositions - Board.RowLength + 1
55 when direction is Direction.LowerLeft or Direction.LowerRight:
56 case >= Board.RowLength and <= Board.NbPositions - Board.RowLength + 1
57 when pos % (Board.RowLength * 2) == Board.RowLength + 1 &&
58 direction is Direction.LowerLeft or Direction.UpperLeft ||
59 pos % (Board.RowLength * 2) == Board.RowLength &&
60 direction is Direction.LowerRight or Direction.UpperRight:
61 return -1;
62 }
63
64 // Return the according position
65 return pos + (int) direction - (pos - 1) / Board.RowLength % 2;
66 }
